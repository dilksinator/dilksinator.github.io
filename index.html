document.getElementById('loadBtn').addEventListener('click', () => { fetch('idylls.txt') .then(data => { const lines = data.split(/\r?\n/).filter(line => line.trim() !== ''); const randomLine = lines[Math.floor(Math.random() * lines.length)]; document.getElementById('output').textContent = randomLine; }) .catch(error => { console.error('Error fetching file:', error); document.getElementById('output').textContent = 'Failed to load file.'; }); });
